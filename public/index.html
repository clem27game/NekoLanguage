// Créer un site web
neksite.créer, script {
    contenu : ("Bienvenue sur mon site NekoScript!"),
    titre : "Mon Premier Site",
    lang : "fr",
    couleur-de-fond : "#f0f0f0",
    style {
        corps : "font-family: Arial, sans-serif",
        titres : "color: #333",
        paragraphe : "line-height: 1.6"
    }
    script {
        nekAfficher("Site chargé!");
    }
}

// Exemple de jeu simple
nekImporter("NekoGame");

joueur = nekCreerSprite("chat.png", 100, 300, 64, 64);
obstacle = nekCreerSprite("arbre.png", 500, 300, 100, 120);
score = 0;

nekJeu.chaqueTrame(function() {
    nekJeu.effacer();
    
    nekSi(nekClavier.toucheAppuyee("droite")) {
        joueur.x = joueur.x plus 5;
    }
    
    nekSi(nekDetecterCollision(joueur, obstacle)) {
        nekJeu.texte("Game Over!", 350, 300, "40px Arial", "#F00");
    }
});

// Bot Discord simple
nekImporter("Discord.neko");

bot = nekDiscord.creer({
    token: "MON_TOKEN_SECRET"
});

nekDiscord.surMessage(function(message) {
    nekSi(message.contenu est "!bonjour") {
        nekDiscord.repondre(message, "Bonjour, " + message.auteur + "!");
    }
});

bot.connecter();

Introduction à NekoScript

NekoScript est un langage de programmation en français conçu pour être simple et accessible. Il permet de créer des sites web, des jeux, et des bots Discord avec une syntaxe intuitive et des fonctions en français.

Les fichiers NekoScript portent l'extension .neko et peuvent être exécutés avec la commande $neko-script exécuter fichier.neko.
